Problem1¶

Two factor ho lee model¶

\begin{equation} r(t) = x_1(t) + x_2(t) \\ dx_1(t) = \mu_1dt+\sigma_1dW_1(t) \\ dx_2(t) = \mu_2dt+\sigma_2dW_2(t) \\ dW_1(t)dW_2(t) = \rho dt \end{equation}

$L(t,T_1,T_2) = \frac{1}{T2-T1}(\frac{P(t,T1)}{P(t,T2)}-1)$

$ P(t,T) = E[e^{-\int_t^Tr(s)ds}] $

So we want to know the bond dynamics

\begin{equation} x_1(t) = x_1(0)+\int_0^t\mu_1ds+\sigma_1\int_0^tdW_1(s) = x_1(0)+\mu_1t+\sigma_1W_1(t) \\ x_2(t) = x_2(0)+\mu_2t+\sigma_2W_2(t) \\ r(t) = x_1(0)+x_2(0)+(\mu_1+\mu_2)t+\sigma_1W_1(t)+\sigma_2W_2(t) \\ r(T) = x_1(t)+x_2(t)+(\mu_1+\mu_2)(T-t)+\sigma_1(W_1(T)-W_1(t))+\sigma_2(W_2(T)-W_2(t)) \end{equation}

Lemma1: $\int_t^Tr(s)ds$ follows the normal distribution

Prove:

\begin{align} \int_t^Tr(s)ds &= Tr(T)-tr(t)-\int_t^Tsdr(s) \\ &= Tr(T)-tr(t)-\int_t^Ts((\mu1+\mu2)ds+\sigma_1dW_1(s)+\sigma_2dW_2(s)) \\ &= T(r(t)+(\mu_1+\mu_2)(T-t)+\sigma_1(W_1(T)-W_1(t))+\sigma_2(W_2(T)-W_2(t)))-tr(t)-\frac{1}{2}(T^2-t^2)(\mu1+\mu2)-\sigma_1\int_t^TsdW_1(s)-\sigma_2\int_t^TsdW_2(s) \\ &= \frac{1}{2}(T-t)^2(\mu1+\mu2)+(T-t)r(t)+\sigma_1\int_t^T(T-s)dW_1(s)+\sigma_2\int_t^T(T-s)dW_2(s) \end{align}

The sum of joint normal two random variables is normal.

\begin{align} M = E(\int_t^Tr(s)ds) &= \frac{1}{2}(T-t)^2(\mu1+\mu2)+(T-t)r(t)\\ V = Var(\int_t^Tr(s)ds) &= \sigma_1^2\int_t^T(T-s)^2ds + \sigma_2^2\int_t^T(T-s)^2ds +2\rho \sigma_1\sigma_2\int_t^T(T-s)^2ds\\ &= \frac{1}{3}(T-t)^3(\sigma_1^2+\sigma_2^2+2\rho \sigma_1\sigma_2) \\ \mbox{The bond dynamcis are given by} \\ P(t,T) &= E[e^{-\int_t^Tr(s)ds}] \\ &= e^{-M+0.5V} \\ &= e^{-\frac{1}{2}(T^2-t^2)(\mu1+\mu2)-(T-t)r(t)+\frac{1}{6}(T-t)^3(\sigma_1^2+\sigma_2^2+2\rho \sigma_1\sigma_2)} \\ &= e^{-\frac{1}{2}(T^2-t^2)(\mu1+\mu2)+\frac{1}{6}(T-t)^3(\sigma_1^2+\sigma_2^2+2\rho \sigma_1\sigma_2)-(T-t)r(t)} \end{align}

Set \begin{align} A(t,T) &= -\frac{1}{2}(T-t)^2(\mu1+\mu2)+\frac{1}{6}(T-t)^3(\sigma_1^2+\sigma_2^2+2\rho \sigma_1\sigma_2) \ B(t,T) &= T-t \ P(t,T) &= e^{A(t,T)-B(t,T)r(t)} \end{align}

\begin{align} L(t,T_1,T_2) &= \frac{1}{T2-T1}(\frac{P(t,T1)}{P(t,T2)}-1) \\ &= \frac{1}{T2-T1}(\frac{e^{A(t,T1)-B(t,T1)r(t)}}{e^{A(t,T2)-B(t,T2)r(t)}}-1) \\ &= \frac{1}{T2-T1}(e^{(A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)}-1) \\ \end{align}

r(t) follows the normal distribution \begin{align} M = E[(A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)] &= (A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))(r0+(\mu1+\mu2)t) \ V = Var((A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)) &= (B(t,T1)-B(t,T2)^2(\sigma_1^2+\sigma_2^2+2\rho \sigma_1\sigma_2)t\ Futures(0,T1,T2) &= E^Q[L(t,T1,T2)] \ &= E^Q[\frac{1}{T2-T1}(e^{(A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)}-1)] \ &= \frac{1}{T2-T1}(E^Q[e^{(A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)}]-1) \ &= \frac{1}{T2-T1}(e^{M+0.5V}-1) \end{align}

\begin{align} Vol(L(t,T1,T2)) &= \frac{1}{\sqrt{t}}\sqrt{Var(L(t,T1,T2)} \\ &= \frac{1}{\sqrt{t}*(T2-T1)}\sqrt{E(e^{2*(A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)})-E(e^{(A(t,T1)-A(t,T2))-(B(t,T1)-B(t,T2))r(t)})} \\ &= \frac{1}{\sqrt{t}*(T2-T1)}\sqrt{e^{2M+2V}-e^{2M+V}} \end{align}

def A(t,T,para):
    mu1,mu2,sig1,sig2,rho = para
    return -0.5*(T-t)**2*(mu1+mu2)+1./6*(T-t)**3*(sig1**2+sig2**2+2*rho*sig1*sig2)

def B(t,T):
    return T-t

def mean_r(t,r0,para,dist = 'Q'): 
    mu1,mu2,sig1,sig2,rho = para
    m = r0+(mu1+mu2)*t
    if (dist == 'F'):
        m -= (sig1**2+sig2**2+2*rho*sig1*sig2) * pow(t,2)*0.5
    return m

#3 para
def var_r(t,T,r0,para):
    sig1,sig2,rho = para
    return (sig1**2+sig2**2+2*rho*sig1*sig2)*(T-t)
    
def zero_bond(t,T,r_t0,para):
    return np.exp(A(t,T,para)-B(t,T,para)*r_t0)

def fwd_rate(t,T1,T2,r0,para,dist = 'Q'):
    mu1,mu2,sig1,sig2,rho = para
    m = (A(t,T1,para)-A(t,T2,para))-(B(t,T1)-B(t,T2))*mean_r(t,r0,para,dist)
    v = var_r(0,t,r0,para[2:])*(B(t,T1)-B(t,T2))**2
    return 1./0.25*(np.exp(m+v/2.)-1)

#moment generation function is exp(tm+0.5sig^2t^2)
def mgf(t,T1,T2,r0,para,k,dist = 'Q'):
    A_diff = A(t,T1,para) - A(t,T2,para)
    B_diff = B(t,T1) - B(t,T2)
    m = A_diff - B_diff * mean_r(t,r0,para,dist)
    v = var_r(0,t,r0,para[2:])*(B(t,T1)-B(t,T2))**2
    return np.exp(k*(2*m+k*v)/2.)

def fwd_rate_mean(t,T1,T2,r0,para,dist='Q',acrual=0.25):
    m1 = mgf(t,T1,T2,r0,para,1,dist)
    return 1/acrual*(m1-1)

def fwd_rate_sig(t,T1,T2,r0,para,dist='Q',acrual=0.25):
    m2 = mgf(t,T1,T2,r0,para,2,dist)
    m1 = mgf(t,T1,T2,r0,para,1,dist)
    e2 = m2 - m1**2
    return 1/acrual*pow(e2,0.5)

def joint_fit(func1,func2,p0,tol = 1e-8,maxIter = 1000,**kwargs):
    mu1,mu2,sig1,sig2,rho = p0
    p0 = np.array(p0)
    f = lambda x:func1((x[0],x[1],sig1,sig2,rho),dist = 'F')
    g = lambda x:func2((mu1,mu2,x[0],x[1],x[2]),dist='F')
    for i in range(maxIter):
        sig1,sig2,rho = leastsq(g,x0=(sig1,sig2,rho))[0]
        mu1,mu2 = leastsq(f,x0=[mu1,mu2])[0]
        #print mu1,mu2,sig1,sig2,rho
        p1 = np.array([mu1,mu2,sig1,sig2,rho]) 
        if np.sum((p1-p0)**2)<tol:
            print "Success"
            return p1
        else:
            i+=1
            p0 = p1
    print "Fail to converge"       
    return p1
